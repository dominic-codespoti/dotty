# Root-level script to run nvim and insert glyphs (do NOT use -u NONE)
# commands:
#   send: <string>   (supports \n, \uXXXX escapes)
#   sleep: <milliseconds>

# Start neovim normally
send: nvim\n
sleep: 2000

# Put a few glyphs into the buffer
send: :put =nr2char(0xE0B0)\n
sleep: 400
send: :put =nr2char(0xE0B4)\n
sleep: 400
send: :put =nr2char(0xF444)\n
sleep: 400
send: :put =nr2char(0xF0A9)\n
sleep: 400

# Quit Neovim
send: :qa!\n
sleep: 400
